# 将以下内容添加到你现有的Caddyfile中

mlyr.top:4099 {
    reverse_proxy localhost:5099
    
    header {
        Access-Control-Allow-Origin "https://mlyr.top"
        Access-Control-Allow-Methods "GET, POST, OPTIONS"
        Access-Control-Allow-Headers "Content-Type"
    }
    
    @options {
        method OPTIONS
    }
    respond @options 204
}
